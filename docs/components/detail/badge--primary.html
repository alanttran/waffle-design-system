@mixin badge($text, $bg){
  text-transform: uppercase;
  display: flex;
  justify-content: center;
  align-items: center;
  height: 20px;
  font-size: 10px;
  font-weight: 700;
  border-radius: 8px;
  padding: 4px 8px;
  color: $text;
  background: $bg;
}
